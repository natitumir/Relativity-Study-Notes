% Ordinary derivatives

% Partial derivatives
\newcommand{\pp}[2]{\frac{\partial #1}{\partial #2}}

\newcommand{\ppn}[3]{\frac{\partial^{#3} #1}{\partial #2^{#3}}}

% Mixed partial derivatives
\newcommand{\ppm}[3]{\frac{\partial^2 #1}{\partial #2 \partial #3}}

% Covariant derivative
\newcommand{\cov}[2]{\nabla_{#2} #1}

% Lie derivative
\newcommand{\Lie}[1]{\mathcal{L}_{#1}}

% Prime derivative shorthand
\newcommand{\pr}[1]{#1^{\prime}}
\newcommand{\ppr}[1]{#1^{\prime\prime}}

\newcommand{\lera}[1]{\left( #1 \right)}

%\cris{}{} - Christoffel symbols
\newcommand{\cris}[2]{\Gamma^{#1}_{#2}}



% \riemann{m}{n}{p}{q}{r}  — 5th arg is the dummy index
\newcommand{\riemann}[5]{
    \partial_{#3} \cris{#1}{#2#4} 
    - \partial_{#4} \cris{#1}{#2#3} 
    + \cris{#1}{#5#3} \cris{#5}{#2#4} 
    - \cris{#1}{#5#4} \cris{#5}{#2#3}
}

% \ricci{a}{b}{s}{r}  — 3rd arg is contraction dummy, 4th is internal dummy
\newcommand{\ricci}[4]{
    \partial_{#3} \cris{#3}{#1#2} 
    - \partial_{#2} \cris{#3}{#1#3} 
    + \cris{#3}{#4#3} \cris{#4}{#1#2} 
    - \cris{#3}{#4#2} \cris{#4}{#1#3}
}
